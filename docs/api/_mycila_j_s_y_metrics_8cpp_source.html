7float Mycila::JSY::Metrics::thdi(float phi) const {
8 if (powerFactor == 0)
9 return NAN;
10 const float cosPhi = phi == 1 ? 1 : cos(phi);
11 return sqrt((cosPhi * cosPhi) / (powerFactor * powerFactor) - 1);
12}

54Mycila::JSY::Metrics& Mycila::JSY::Metrics::operator+=(const Mycila::JSY::Metrics& other) {
55 // voltage and frequency are not aggregated
56 current += other.current;
57 activePower += other.activePower;
58 apparentPower += other.apparentPower;
59 activeEnergy += other.activeEnergy;
60 activeEnergyImported += other.activeEnergyImported;
61 activeEnergyReturned += other.activeEnergyReturned;
62 reactiveEnergy += other.reactiveEnergy;
63 reactiveEnergyImported += other.reactiveEnergyImported;
64 reactiveEnergyReturned += other.reactiveEnergyReturned;
65 apparentEnergy += other.apparentEnergy;
66 powerFactor = apparentPower == 0 ? NAN : abs(activePower / apparentPower);
67 reactivePower = sqrt(apparentPower * apparentPower - activePower * activePower);
68 return *this;
69}
